**File 1: ScrollWisdomApp.swift** (per D-01, D-04)

Replace the entire file with:

```swift
import SwiftUI
import FirebaseCore
import FirebaseCrashlytics

@main
struct ScrollWisdomApp: App {
    @State private var manager = ContentManager()
    @State private var notificationManager = NotificationManager()
    @State private var storeManager = StoreManager()

    init() {
        FirebaseApp.configure()
    }

    var body: some Scene {
        WindowGroup {
            Group {
                if manager.hasCompletedOnboarding {
                    MainTabView()
                } else {
                    OnboardingView()
                }
            }
            .environment(manager)
            .environment(notificationManager)
            .environment(storeManager)
            .preferredColorScheme(.dark)
            .task {
                Crashlytics.crashlytics().setCustomValue(
                    storeManager.isPremium,
                    forKey: "is_premium"
                )
            }
            .onChange(of: storeManager.isPremium) { _, newValue in
                Crashlytics.crashlytics().setCustomValue(newValue, forKey: "is_premium")
            }
        }
    }
}
```

Key changes:
- Added `import FirebaseCore` and `import FirebaseCrashlytics`
- Added `init()` with `FirebaseApp.configure()` (per D-01)
- Added `.task` modifier to set initial isPremium custom value (per D-04)
- Added `.onChange(of: storeManager.isPremium)` to update custom value when subscription status changes (per D-04)
